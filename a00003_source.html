41 /// A record consisting of string fields separated by a delimiter. The functionality
42 /// is similar to the split() function in Perl or the way awk treats lines. Initialize
43 /// the class with a string and access the fields by the operator[].
44 class SfiDelimitedRecordSTD
45 {
46 protected:
47  /// Buffer with a modified string for fast retrieval.
48  string m_buffer;
49  char m_delimiter;
50  SfiVectorLite<int> m_offsets;
51  SfiVectorLite<int> m_lengths;
52  char* m_sptr;
53  const char m_nullChar;
54 
55 public:
56  explicit SfiDelimitedRecordSTD(const char* str = 0, char delimiter = ',') : m_delimiter(delimiter), m_sptr(0), m_nullChar(0)
57  {

59  m_lengths.reserve(6);
60  *this = str;
61  }

100  /// Returns the i-th field or an empty string if there are fewer than i fields.
101  const char* operator[](int i) const
102  {

104  if (size == 0 || i < 0 || i >= size)

129  /// Returns the number of fields in the \c buf. Delimiters inside double quotes are ignored.
130  /// \c n is the size of string in \c buf, excluding the terminating null.
131  /// NOTE that buf's delimiters get overwritten with null characters.
132  /// Access to fields is provided by \c get(int).
133  int split(char* buf, int n)
134  {
135  if (!buf)
136  {
137  clear();
138  return 0;
139  }
140  m_sptr = buf;
141  int start = 0;
142  int i;
143  m_offsets.clear();
144  m_lengths.clear();
145  bool insideQuotes = false;
146  for (i = 0; i < n; i++)
147  {
148  if (buf[i] == '"')
149  {
150  insideQuotes = !insideQuotes;
151  }
152  if (!insideQuotes && buf[i] == m_delimiter)
153  {
154  buf[i] = '\0';
155  m_offsets.push_back(start);
156  m_lengths.push_back(i - start);
157  start = i + 1;
158  }
159  }
160  m_offsets.push_back(start);
161  m_lengths.push_back(i - start);
162  return i ? m_offsets.size() : 0;
163  }
164 
165  /// Returns a pointer to the i-th field of a split string - for use with split(char*, int) only!!!
166  /// If the index i is outside the range of valid fields, a pointer to a null character is returned.
167  const char* get(int i) const
168  {
169  int size = m_offsets.size();
170  if (size == 0 || i < 0 || i >= size)
171  {
172  return &m_nullChar; //m_sptr + m_offsets[size - 1] + m_lengths[size - 1];
173  }
174  return m_sptr + m_offsets[i];
175  }

179  /// Returns the number of fields in the record. Delimiters inside double quotes are ignored.
180  int split()
181  {
182  int start = 0;
183  int i, n;
184  m_offsets.clear();
185  m_lengths.clear();
186  bool insideQuotes = false;
187  for (i = 0, n = m_buffer.length(); i < n; i++)
188  {
189  if (m_buffer[i] == '"')
190  {
191  insideQuotes = !insideQuotes;
192  }
193  if (!insideQuotes && m_buffer[i] == m_delimiter)
194  {
195  m_buffer[i] = '\0';
196  m_offsets.push_back(start);
197  m_lengths.push_back(i - start);
198  start = i + 1;
199  }
200  }
201  m_offsets.push_back(start);
202  m_lengths.push_back(i - start);
203  return i ? m_offsets.size() : 0;
204  }

206  /// Clears the record.
207  void clear()
208  {
209  m_buffer.clear();
210  m_offsets.clear();
211  m_lengths.clear();
212  m_sptr = 0;
213  }
